def display_menu():
    print("\n===== Student Grade Management System =====")
    print("1. Add student")
    print("2. Display all students")
    print("3. Search student by name")
    print("4. Exit")

students = []

def add_student():
    name = input("Enter student name: ")
    try:
        grade = float(input("Enter student grade (0-100): "))
        if 0 <= grade <= 100:
            students.append({"name": name, "grade": grade})
            print(f"{name} added successfully!")
        else:
            print("Grade must be between 0 and 100.")
    except ValueError:
        print("Invalid input. Grade must be a number.")

def display_students():
    if not students:
        print("No students added yet.")
    else:
        print("\n--- Student List ---")
        for idx, student in enumerate(students, 1):
            print(f"{idx}. {student['name']} - Grade: {student['grade']}")

def search_student():
    search_name = input("Enter student name to search: ").lower()
    found = False
    for student in students:
        if student["name"].lower() == search_name:
            print(f"Found: {student['name']} - Grade: {student['grade']}")
            found = True
            break
    if not found:
        print("Student not found.")

# Main loop
while True:
    display_menu()
    choice = input("Enter your choice (1-4): ")

    if choice == "1":
        add_student()
    elif choice == "2":
        display_students()
    elif choice == "3":
        search_student()
    elif choice == "4":
        print("Exiting program. Goodbye!")
        break
    else:
        print("Invalid choice. Please enter 1 to 4.")
